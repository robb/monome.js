monome.js

A lightweight library for writing monome applications with node.

Monome is a company that produces adaptable, minimalist interfaces, such as the sixty four, the two fifty-six and the arc.
Find them at monome.org.

Monome.js is a lightweight wrapper around libmonome, it aims at simplifying the requirements for writing beatiful monome applications.

an example.

{Monome} = require 'monome'
monome = new Monome '/dev/tty.usbserial-m64-0123'

monome.onbuttondown = ({x, y}) ->
  monome.set x, y, on

monome.onbuttonup = ({x, y}) ->
  monome.set x, y, off

monome.start()

Please note this example uses CoffeeScript rather than JavaScript.
